const blobMap = new Map(); const urlMap = new Map(); // --- THE CRITICAL FIX: FORCING THE ENGINE TO SEE OUR UNZIPPED FILES --- const OriginalImage = window.Image; window.Image = function() { const img = new OriginalImage(); const setter = Object.getOwnPropertyDescriptor(HTMLImageElement.prototype, 'src').set; Object.defineProperty(img, 'src', { set: function(val) { // Get just the filename (e.g., "M0001.png") const filename = val.toString().split('/').pop(); // If we have this file in our unzipped memory, use our Blob URL instead if (urlMap.has(filename)) { setter.call(this, urlMap.get(filename)); } else { setter.call(this, val); } } }); return img; }; // Hook Fetch & XHR (Backups for other data types) const ogFetch = window.fetch; window.fetch = async (input, init) => { const url = typeof input === 'string' ? input : input.url; const name = url.split('/').pop(); if (blobMap.has(name)) return new Response(blobMap.get(name)); return ogFetch(input, init); }; const ogOpen = XMLHttpRequest.prototype.open; XMLHttpRequest.prototype.open = function(m, url) { const name = url.toString().split('/').pop(); if (urlMap.has(name)) url = urlMap.get(name); return ogOpen.apply(this, arguments); }; async function init() { try { let chunks = []; // 1. Download the parts for (let i = 1; i <= 20; i++) { document.getElementById('percentage').innerText = Math.floor((i/20)*40) + '%'; document.getElementById('bar').style.width = Math.floor((i/20)*40) + '%'; const res = await ogFetch(`resources.zip.part${i}`); if (!res.ok) throw new Error("Missing part " + i); chunks.push(await res.arrayBuffer()); } // 2. Combine and Unzip const zip = await JSZip.loadAsync(new Blob(chunks)); const files = Object.keys(zip.files); for (let i = 0; i < files.length; i++) { const file = zip.files[files[i]]; if (file.dir) continue; const name = files[i].split('/').pop(); const data = await file.async("uint8array"); // Create a local URL for the file const blob = new Blob([data], { type: name.endsWith('.png') ? 'image/png' : 'application/octet-stream' }); blobMap.set(name, blob); urlMap.set(name, URL.createObjectURL(blob)); // Update loading bar (remaining 60%) document.getElementById('percentage').innerText = (40 + Math.floor((i/files.length)*60)) + '%'; document.getElementById('bar').style.width = (40 + Math.floor((i/files.length)*60)) + '%'; } document.getElementById('loader-ui').style.display = 'none'; document.getElementById('play-button').style.display = 'block'; } catch (e) { console.error(e); document.getElementById('stage').innerText = "ERROR: " + e.message; } } document.getElementById('play-button').onclick = () => { // Unlock Audio for browsers const AC = window.AudioContext || window.webkitAudioContext; if (AC) new AC().resume(); document.getElementById('selection').style.display = 'none'; document.getElementById('MMFCanvas').style.display = 'block'; const script = document.createElement('script'); script.src = "Runtime.js"; script.onload = () => { // Find the .cch file in our unzipped list and start const cchFile = Array.from(urlMap.keys()).find(k => k.toLowerCase().endsWith('.cch')); if (cchFile) { new Runtime("MMFCanvas", cchFile); } else { alert("Error: .cch game file not found in zip!"); } }; document.head.appendChild(script); }; window.onload = init;
